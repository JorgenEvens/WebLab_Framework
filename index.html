Web-lab framework

Web-lab's web application framework

Download this project as a .zip file Download this project as a tar.gz file

Downloading source

$ git clone git://github.com/NoSTaBoNN/WebLab_Framework WebLab

Usage

index.php

<?php
include( 'WebLab/Framework.php' );

$app = new WebLab_Framework( 'config.json' );
$app->run();

Application/Loader.php

<?php
    class Loader extends WebLab_Loader_Application
    {
        protected function environment() {
            session_start();

            WebLab_Template::setRootTemplate( new WebLab_Template( 'main/index' ) );

        }

        protected function controlDispatcher() {
            $dispatcher = new WebLab_Dispatcher_Visit();
            $dispatcher->execute();
        }

        protected function shutdown() {
            echo WebLab_Template::getRootTemplate();
        }
    }
